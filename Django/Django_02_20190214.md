```python
from datetime import datetime

def template_example(request):
    my_list = ['pasta', 'pizza', 'steak', 'taco']
    my_sentence = 'Life is short, you need python'
    messages = ['apple', 'banana', 'cucumber', 'mango']
    empty_list = []
    datetimenow = datetime.now()
    return render(request, 'template_example.html', 
                {'my_list': my_list, 'my_sentence': my_sentence,
                    'messages': messages, 'empty_list': empty_list,
                    'datetimenow': datetimenow
                })
```

<body style='height: 10000px;'>
    <h3>1. 반복문</h3>
    {% for menu in my_list %}
        <p>{{ menu }}</p>
    {% endfor %}
    <hr>
    {% for menu in my_list %}
        <!-- 순서를 출력해줌 -->
        {{ forloop.counter }}
        <!-- 루프 내 첫번째의 출력을 바꿈 / last -->
        {% if forloop.first %}
            <p>pasta+tomato</p>
        {% else %}
            <p>{{ menu }}</p>
        {% endif %}
    {% endfor %}
    <hr>
    <!-- 값이 없으면 empty 부분 출력 -->
    {% for user in empty_list %}
        <p>{{ user }}</p>
    {% empty %}
        <p>지금 가입한 회원이 없습니다.</p>
    {% endfor %}
    <hr>
    
    <h3>2. 조건문</h3>
    {% if 'pasta' in my_list %}
        <p>pasta needs more sauce</p>
    {% endif %}
    <hr>
    
    <h3>3. length filter 활용( | 파이프라인)</h3>
    {% for message in messages %}
        {% if message|length > 5 %}
            <p>tooooooo long</p>
        {% else %}
            <p>{{ message }}, {{ message|length }}</p>
        {% endif %}
    {% endfor %}
    <hr>
    
    <h3>4. lorem ipsum</h3>
    {% lorem %} <!-- 내장변수라서 % 붙여줌 -->
    <hr>
    {% lorem 3 w %} <!-- 3 단어만 출력 -->
    <hr>
    {% lorem 4 w random %} <!-- 새로고침할 때마다 랜덤한 4개의 단어를 출력 -->
    <hr>
    {% lorem 2 p %} <!-- 2개 문단 출력 -->
    <hr>
    
    <h3>5. 글자수 제한 (끝을 자른 후 /공백.../ 을 붙여서 출력 - 기본 값으로 4개의 캐릭터를 먹고 있음)</h3>
    <p>{{ my_sentence|truncatewords:3 }}</p>
    <p>{{ my_sentence|truncatechars:6 }}</p> <!-- 4캐릭터를 먹고있기 때문에 출력하려는 캐릭터수 + 4로 지정해줘야 함 -->
    <p>{{ my_sentence|truncatechars:10 }}</p>
    <hr>
    
    <h3>6. 글자 관련 filter</h3>
    <p>{{ 'abc'|length }}</p>
    <p>{{ 'ABC'|lower }}</p>
    <p>{{ my_sentence|title }}</p>
    <p>{{ 'abc def'|capfirst }}</p>
    <p>{{ my_list|random }}</p>

    <h3>7. 연산</h3>
    <p>{{ 4|add:6 }}</p>
    <hr>

    <hr>
